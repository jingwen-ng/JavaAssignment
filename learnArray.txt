import java.util.ArrayList;
import java.util.Arrays;
import java.util.Collections;
import java.util.Comparator;

/**
 * LearnArray
 */
public class LearnArray {

  static void remove(ArrayList<Integer> y) {
    y.remove(2);
    System.out.println(y);
  }

  static void sort(ArrayList<Integer> y) {
    y.sort(Comparator.naturalOrder());
    System.out.println(y);
  }

  static Integer find(ArrayList<Integer> y) {
    return y.indexOf(7);
  }

  static Boolean check(ArrayList<Integer> y) {
    return y.contains(4);
  }

  public static void main(String[] args) {
    Integer[] x = {8, 5, 1, 2, 7, 4};

    // convert to ArrayList
    ArrayList<Integer> y = new ArrayList<>(Arrays.asList(x));

    // remove the 3rd number
    remove(y);
    // print the sorted ArrayList
    sort(y);

    // print the index of the number 7
    // find an element with a certain value, and get its index
    find(y);

    // print boolean representing if the ArrayList contains the number 4
    check(y);

    // print the element at index 4
    System.out.println(y.get(4));

    // change the element at index 2 into value: 123
    y.set(2, 123);
    System.out.println(y);
  }
}
